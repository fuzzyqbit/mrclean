export interface Finding {
  ruleId: string
  severity: 'CRITICAL' | 'HIGH' | 'MEDIUM' | 'LOW'
  span: { start: number; end: number }
  value: string
  redactedHash: string
  fingerprint: string
  source: 'secretlint' | 'gitleaks' | 'entropy' | 'env' | 'words' | 'pii-regex' | 'pii-ner'
  action?: 'block' | 'substitute' | 'audit' | 'off' | 'warn'
}
export function redactedHash(value: string): string   // 16-hex
export function fingerprint(ruleId: string, value: string): string  // `${ruleId}:${redactedHash}`
// spansOverlap predicate (internal, half-open [start,end)): a.span.start < b.span.end && b.span.start < a.span.end
