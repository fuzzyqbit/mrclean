<objective>
Implement `mrclean doctor`: the operator's "is it actually working?" command. Reads the install state from `~/.claude/settings.json` + `~/.claude.json`, verifies absolute paths are still executable, round-trips a seeded canary string through both the hook bin (synthetic stdin/stdout) AND the MCP server (initialize + `sanitize` call), exercises the three-layer config reader from Plan 01-02b, reads the Claude Code version, and emits a PASS/FAIL report with structured exit codes (RESEARCH §4.4).

Purpose: This is Phase 1 success criterion #2 — the operator-runnable proof that the wiring is alive. After this plan, `mrclean install && mrclean doctor` is a complete end-to-end demo of Phase 1. Plus, doctor is the verification harness Plan 02/03/04 acceptance criteria will use in the future. The config-load check makes CFG-01/CFG-03 observable to the operator as a bonus.

Output: A robust doctor command that satisfies INST-06 and turns the four prior plans' work plus the config reader into a single one-command operator-visible PASS. The orchestrator lives in two functions: `computeDoctorReport` (pure — returns `{ exitCode, results, versionResult }`, never calls `process.exit`) so vitest can call it directly, and `runDoctor` (CLI wrapper — calls computeDoctorReport then `process.exit`).
</objective>

      import { spawnSync } from 'node:child_process';
      export const CANARY_STRING = 'MRCLEAN_CANARY_PHASE1_DOCTOR_TEST';

      export async function runHookCanary(nodePath: string, mrcleanBin: string): Promise<{ ok: boolean; detail: string }> {
        const payload = JSON.stringify({
          hook_event_name: 'UserPromptSubmit',
          session_id: 'doctor-canary',
          transcript_path: '/tmp/doctor-canary',
          cwd: process.cwd(),
          prompt: `doctor canary: ${CANARY_STRING}`,
        });
        const result = spawnSync(nodePath, [mrcleanBin, 'hook'], { input: payload, encoding: 'utf8', timeout: 5_000 });
        if (result.status !== 0) {
          return { ok: false, detail: `hook exited ${result.status}: ${result.stderr.split('\\n')[0]}` };
        }
        try {
          const parsed = JSON.parse(result.stdout);
          const ctx = parsed?.hookSpecificOutput?.additionalContext ?? '';
          if (typeof ctx !== 'string' || !ctx.startsWith('mrclean active v')) {
            return { ok: false, detail: `hook stdout did not contain wiring banner: ${result.stdout.slice(0, 100)}` };
          }
          return { ok: true, detail: 'hook canary round-tripped; wiring banner present' };
        } catch (err) {
          return { ok: false, detail: `hook stdout not valid JSON: ${result.stdout.slice(0, 100)}` };
        }
      }

      export async function runMcpCanary(nodePath: string, mcpBin: string): Promise<{ ok: boolean; detail: string }> {
        const { Client } = await import('@modelcontextprotocol/sdk/client/index.js');
        const { StdioClientTransport } = await import('@modelcontextprotocol/sdk/client/stdio.js');
        const transport = new StdioClientTransport({ command: nodePath, args: [mcpBin] });
        const client = new Client({ name: 'mrclean-doctor', version: '0.0.0' }, { capabilities: {} });
        try {
          await client.connect(transport);
          const result = await client.callTool({ name: 'sanitize', arguments: { text: CANARY_STRING } });
          const echoed = (result.content as Array<{ type: string; text: string }>)?.[0]?.text;
          if (echoed !== CANARY_STRING) {
            return { ok: false, detail: `MCP canary did not round-trip; got: ${String(echoed).slice(0, 100)}` };
          }
          return { ok: true, detail: 'MCP canary round-tripped through sanitize tool' };
        } catch (err) {
          return { ok: false, detail: `MCP client error: ${err instanceof Error ? err.message : String(err)}` };
        } finally {
          await client.close().catch(() => {});
        }
      }

      import { homedir } from 'node:os';
      import { join } from 'node:path';
      import { checkHooksRegistered, checkMcpRegistered, checkBinsExecutable, checkHookCanary, checkMcpCanary, checkConfigLoad, type CheckResult } from './checks.js';
      import { checkClaudeCodeVersion, type ClaudeVersionResult } from './version-check.js';
      import { renderReport, computeExitCode } from './report.js';
      import { resolveNodePath, resolveMrcleanBinPath, resolveMrcleanMcpPath } from '../install/path-resolver.js';

      export interface DoctorOpts { homeDir?: string; cwd?: string }
      export interface DoctorReport {
        exitCode: number;
        results: CheckResult[];
        versionResult: ClaudeVersionResult;
      }

      /**
       * Pure (testable) core. Runs every check, computes the final exit code, returns the full report.
       * Does NOT call process.exit. Callers (the CLI wrapper, or vitest E2E tests) decide what to do.
       *
       * The `quiet` flag is reserved for tests that want to skip rendering — runDoctor calls
       * renderReport AFTER this function returns, so this function never writes to stdout.
       */
      export async function computeDoctorReport(opts: { homeDir: string; cwd: string; quiet?: boolean }): Promise<DoctorReport> {
        const { homeDir, cwd } = opts;
        const settingsPath = join(homeDir, '.claude/settings.json');
        const claudeJsonPath = join(homeDir, '.claude.json');

        const results: CheckResult[] = [];
        results.push(await checkHooksRegistered(settingsPath));
        results.push(await checkMcpRegistered(claudeJsonPath, cwd));

        const nodePath = resolveNodePath();
        const mrcleanBin = await resolveMrcleanBinPath().catch(() => '<unresolved>');
        const mcpBin = await resolveMrcleanMcpPath().catch(() => '<unresolved>');
        results.push(await checkBinsExecutable(settingsPath, claudeJsonPath, cwd));

        // Only attempt canary if bins are executable, otherwise skip with explanatory result
        const binsResult = results[results.length - 1]!;
        if (binsResult.status === 'PASS') {
          results.push(await checkHookCanary(nodePath, mrcleanBin));
          results.push(await checkMcpCanary(nodePath, mcpBin));
        } else {
          results.push({ name: 'hook-canary', status: 'SKIP', detail: 'skipped: bins not executable', exitCodeOnFail: 4 });
