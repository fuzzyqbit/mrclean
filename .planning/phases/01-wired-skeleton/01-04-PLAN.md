      ```typescript
      export function installShutdownHandlers(closeFn: () => Promise<void>): void {
        let shuttingDown = false;
        const handler = (signal: NodeJS.Signals) => {
          if (shuttingDown) return;
          shuttingDown = true;
          process.stderr.write(`mrclean-mcp: received ${signal}, shutting down\n`);
          closeFn()
            .then(() => process.exit(0))
            .catch((err) => {
              process.stderr.write(`mrclean-mcp: shutdown error: ${err instanceof Error ? err.message : String(err)}\n`);
              process.exit(1);
            });
        };
        process.on('SIGINT', () => handler('SIGINT'));
        process.on('SIGTERM', () => handler('SIGTERM'));
      }
      ```
    Create `src/mcp/server.ts` (replacing the Plan 01 stub):
      ```typescript
      import { VERSION } from '../shared/version.js';
      import { installShutdownHandlers } from './lifecycle.js';

      export async function runMcpServer(): Promise<void> {
        // Lazy-import per RESEARCH §6.2 — keeps CLI cold-start cheap.
        const { McpServer } = await import('@modelcontextprotocol/sdk/server/mcp.js');
        const { StdioServerTransport } = await import('@modelcontextprotocol/sdk/server/stdio.js');
        const server = new McpServer({ name: 'mrclean', version: VERSION });

        const { registerSanitizeTool } = await import('./tools/sanitize.js');
        const { registerRestoreTool } = await import('./tools/restore.js');
        const { registerAuditQueryTool } = await import('./tools/audit-query.js');
        registerSanitizeTool(server);
        registerRestoreTool(server);
        registerAuditQueryTool(server);

        const transport = new StdioServerTransport();
        // SIGINT/SIGTERM are registered EXACTLY ONCE via installShutdownHandlers below.
        // The event loop is held open by the transport's stdio read loop + server.connect lifecycle —
        // we do NOT register additional signal listeners here (avoids MaxListenersExceededWarning
        // and the SIGINT exit race that would otherwise occur if both installShutdownHandlers AND
        // a Promise-based SIGINT/SIGTERM listener tried to handle the same signal).
        installShutdownHandlers(async () => { await transport.close(); });
        await server.connect(transport);
        process.stderr.write(`mrclean-mcp v${VERSION} running on stdio\n`);
        // server.connect() returns once the transport is wired; the stdio transport keeps reading from
        // stdin until EOF or until installShutdownHandlers fires process.exit(0) on signal. No further
        // event-loop-keeping construct is required — and intentionally none is added (see comment above).
      }
      ```
